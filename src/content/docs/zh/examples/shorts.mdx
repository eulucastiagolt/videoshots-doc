      controls: 1
    });

  </script>

## React 中的短视频

```jsx
import { useEffect, useRef } from 'react';
import VideoShorts from 'videoshots';
import 'videoshots/dist/videoshots.min.css';

function ShortsFeed({ shorts }) {
  const containerRef = useRef(null);
  const instanceRef = useRef(null);

  useEffect(() => {
    instanceRef.current = new VideoShorts(containerRef.current, shorts, {
      lazy: true,
      width: '280px',
      aspectRatio: '9/16',
      muted: true,
      loop: true,
      controls: 1
    });

    instanceRef.current.init();

    return () => {
      instanceRef.current?.destroy();
    };
  }, [shorts]);

  const handlePlay = () => instanceRef.current?.play();
  const handlePause = () => instanceRef.current?.pause();
  const handleMute = () => instanceRef.current?.mute();
  const handleUnmute = () => instanceRef.current?.unMute();

  return (
    <div>
      <div style={{ display: 'flex', gap: '10px', justifyContent: 'center', marginBottom: '20px' }}>
        <button onClick={handlePlay}>全部播放</button>
        <button onClick={handlePause}>全部暂停</button>
        <button onClick={handleMute}>静音</button>
        <button onClick={handleUnmute}>取消静音</button>
      </div>
      <div ref={containerRef} data-videoshort />
    </div>
  );
}

export default ShortsFeed;
```
