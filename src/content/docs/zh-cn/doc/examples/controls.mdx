## 音量滑块控制

```js
const videos = new VideoShorts('#container', videoUrls, options);
await videos.init();

// 音量滑块
const volumeSlider = document.getElementById('volume-slider');
const volumeValue = document.getElementById('volume-value');

volumeSlider.addEventListener('input', (e) => {
  const volume = parseInt(e.target.value);
  videos.setVolume(volume);
  volumeValue.textContent = volume;
});

// 静音按钮
document.getElementById('mute-btn').addEventListener('click', () => {
  videos.mute();
});

document.getElementById('unmute-btn').addEventListener('click', () => {
  videos.unMute();
});
```

## 进度控制

```html
<div class="progress-controls">
  <input type="range" id="progress" min="0" max="100" value="0">
  <span id="time-display">0:00 / 0:00</span>
</div>

<script>
const progress = document.getElementById('progress');
const timeDisplay = document.getElementById('time-display');

// 每秒更新进度
setInterval(() => {
  const currentTime = videos.getCurrentTime(0);
  const duration = videos.getDuration(0);
  
  if (currentTime && duration) {
    const percent = (currentTime / duration) * 100;
    progress.value = percent;
    
    const currentFormatted = formatTime(currentTime);
    const durationFormatted = formatTime(duration);
    timeDisplay.textContent = `${currentFormatted} / ${durationFormatted}`;
  }
}, 1000);

// 滑块变化时跳转
progress.addEventListener('input', (e) => {
  const duration = videos.getDuration(0);
  const seekTime = (e.target.value / 100) * duration;
  videos.seekTo(seekTime, 0);
});

function formatTime(seconds) {
  const mins = Math.floor(seconds / 60);
  const secs = Math.floor(seconds % 60);
  return `${mins}:${secs.toString().padStart(2, '0')}`;
}
</script>
```

## 键盘控制

```js
document.addEventListener('keydown', (e) => {
  switch(e.key) {
    case ' ':
      e.preventDefault();
      // 切换当前视频的播放/暂停
      if (videos.isPlaying(0)) {
        videos.pause(0);
      } else {
        videos.play(0);
      }
      break;
    case 'm':
      videos.toggleMute(0);
      break;
    case 'ArrowLeft':
      const currentTime = videos.getCurrentTime(0) || 0;
      videos.seekTo(Math.max(0, currentTime - 10), 0);
      break;
    case 'ArrowRight':
      const current = videos.getCurrentTime(0) || 0;
      videos.seekTo(current + 10, 0);
      break;
    case 'ArrowUp':
      const vol = videos.getVolume(0) || 50;
      videos.setVolume(Math.min(100, vol + 10), 0);
      break;
    case 'ArrowDown':
      const volume = videos.getVolume(0) || 50;
      videos.setVolume(Math.max(0, volume - 10), 0);
      break;
  }
});
```

```html
  <div class="custom-player">
    <div id="video-container" data-videoshort></div>
    <div class="custom-controls">
      <div class="progress-bar" onclick="seekToPosition(event)">
        <div class="progress-fill" id="progress-fill"></div>
      </div>
      <div class="control-buttons">
        <button onclick="togglePlay()">▶</button>
        <button onclick="videos.seekTo((videos.getCurrentTime(0) || 0) - 10, 0)">⏪</button>
        <button onclick="videos.seekTo((videos.getCurrentTime(0) || 0) + 10, 0)">⏩</button>
        <span class="time-display" id="time-display">0:00 / 0:00</span>
        <div class="volume-control">
          <button onclick="videos.toggleMute(0)">🔊</button>
          <input type="range" min="0" max="100" value="100" 
                 onchange="videos.setVolume(this.value, 0)">
        </div>
      </div>
    </div>
  </div>

  <script src="https://cdn.jsdelivr.net/gh/eulucastiagolt/videoshorts@1.6.7/dist/videoshorts.min.js"></script>
  <script>
    let videos;
    let playBtn;
    
    async function init() {
      videos = new VideoShorts('#video-container', [
        'https://www.youtube.com/watch?v=SGfb6y1j5x0'
      ], {
        lazy: false,
        width: '640px',
        aspectRatio: '16/9',
        controls: 0,
        modestbranding: 1,
        onStateChange: (event) => {
          playBtn = document.querySelector('.control-buttons button:first-child');
          if (event.data === YT.PlayerState.PLAYING) {
            playBtn.textContent = '⏸';
          } else {
            playBtn.textContent = '▶';
          }
        }
      });
      
      await videos.init();
      
      setInterval(updateProgress, 100);
    }
    
    function togglePlay() {
      if (videos.isPlaying(0)) {
        videos.pause(0);
      } else {
        videos.play(0);
      }
    }
    
    function updateProgress() {
      const current = videos.getCurrentTime(0) || 0;
      const duration = videos.getDuration(0) || 1;
      const percent = (current / duration) * 100;
      
      document.getElementById('progress-fill').style.width = percent + '%';
      
      const timeDisplay = document.getElementById('time-display');
      timeDisplay.textContent = `${formatTime(current)} / ${formatTime(duration)}`;
    }
    
    function seekToPosition(e) {
      const rect = e.target.getBoundingClientRect();
      const percent = (e.clientX - rect.left) / rect.width;
      const duration = videos.getDuration(0) || 0;
      videos.seekTo(percent * duration, 0);
    }
    
    function formatTime(seconds) {
      const mins = Math.floor(seconds / 60);
      const secs = Math.floor(seconds % 60);
      return `${mins}:${secs.toString().padStart(2, '0')}`;
    }
    
    init();
  </script>
</body>
</html>
```
