## With React

```jsx
import { useEffect, useRef } from 'react';
import VideoShorts from 'videoshorts';
import 'videoshorts/dist/videoshorts.min.css';

function VideoGallery({ videoUrls }) {
  const containerRef = useRef(null);
  const instanceRef = useRef(null);

  useEffect(() => {
    if (containerRef.current) {
      instanceRef.current = new VideoShorts(containerRef.current, videoUrls, {
        lazy: true,
        width: '100%',
        aspectRatio: '16/9'
      });
      
      instanceRef.current.init();
    }

    return () => {
      instanceRef.current?.destroy();
    };
  }, [videoUrls]);

  return (
    <div 
      ref={containerRef} 
      data-videoshort 
      style={{ maxWidth: '800px', margin: '0 auto' }}
    />
  );
}

export default VideoGallery;
```

## With Vue 3

```vue
<template>
  <div 
    ref="container" 
    data-videoshort
    style="max-width: 800px; margin: 0 auto;"
  />
</template>

<script setup>
import { ref, onMounted, onUnmounted, watch } from 'vue';
import VideoShorts from 'videoshorts';
import 'videoshorts/dist/videoshorts.min.css';

const props = defineProps({
  videos: {
    type: Array,
    required: true
  }
});

const container = ref(null);
let instance = null;

onMounted(async () => {
  instance = new VideoShorts(container.value, props.videos, {
    lazy: true,
    width: '100%',
    aspectRatio: '16/9'
  });
  
  await instance.init();
});

onUnmounted(() => {
  instance?.destroy();
});

watch(() => props.videos, async (newVideos) => {
  instance?.destroy();
  instance = new VideoShorts(container.value, newVideos, {
    lazy: true,
    width: '100%',
    aspectRatio: '16/9'
  });
  await instance.init();
});
</script>
```

## With Next.js

```jsx
'use client';

import { useEffect, useRef } from 'react';
import dynamic from 'next/dynamic';

function VideoPlayer({ videos }) {
  const containerRef = useRef(null);
  const instanceRef = useRef(null);

  useEffect(() => {
    let VideoShorts;

    import('videoshorts').then((module) => {
      VideoShorts = module.default;
      
      instanceRef.current = new VideoShorts(containerRef.current, videos, {
        lazy: true,
        width: '100%',
        aspectRatio: '16/9'
      });
      
      instanceRef.current.init();
    });

    return () => {
      instanceRef.current?.destroy();
    };
  }, [videos]);

  return <div ref={containerRef} data-videoshort />;
}

export default dynamic(() => Promise.resolve(VideoPlayer), { ssr: false });
```
